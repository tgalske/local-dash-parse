  private String findNextProperty(String targetWord,
          String nextTargetWord,
          int wordIndex) {
    while (!textLines.get(index)[wordIndex].equals(targetWord)) {
      wordIndex++;
    }
    wordIndex++;
    StringBuilder property = new StringBuilder();
    while (!textLines.get(index)[wordIndex].equals(nextTargetWord)) {
      property.append(textLines.get(index)[wordIndex]);
      if (!textLines.get(wordIndex + 1).equals(nextTargetWord)) {
        property.append(" ");
      }
      wordIndex++;
    }
    return property.toString();
  }
